// DLOI T5 魔术数组 
#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin >> n;

    int a[n][n];
    for (int i = 0; i < n; i++)    cin >> a[0][i];
    for (int i = 0; i < n; i++) cin >> a[i][i];

    if (n == 1)
    {
        cout << a[0][0] << endl;
        return 0;
    }

    // 前一行与后一行的所有数之间都有公差 
    int cha;
    for (int i = 1; i < n; i++)
    {
        cha = a[i-1][i]-a[i][i];
        for (int j = 0; j < n; j++)
            a[i][j] = a[i-1][j] - cha;
    }

    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
            cout << a[i][j] << ' ';
        cout << '\n';
    }
    return 0;
}

// DLOI T6 三角形个数
// OnlineJudge: [Mr.Huang OJ] http://111.230.62.117/problem.php?id=1905 
#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin >> n;

    long long cnt = 0;
    for (int i = 0; i < n; i++)
    {
        long long l;
        cin >> l;
        if (l > 0) cnt++;  // 负数不算 
    }

    // 排列
    // 只要是正数都可组成三角形
    // 求 从 n 个整数中取出 3 个数，有多少种排列方案

    long long ans = cnt*(cnt-1)*(cnt-2)/6;
    cout << ans << endl;
    return 0;
}
