	int q;

		int t;

					s.push(!t);

					int t = s.top();
					s.pop();
					int t2 = s.top();
					s.pop();
					s.push(t && t2);

					int t = s.top();
					s.pop();
					int t2 = s.top();

				}
				
				cur.clear();

		}

#include <iostream>
#include <cstdio>
#include <fstream>
#include <algorithm>
using namespace std;
long long ans = -99999999999;
int a[1005][1005], n, m;
bool vis[1005][1005];
const int dx[3] = { -1, 1, 0 };
const int dy[3] = { 0,  0, 1 };
void dfs(int x, int y, long long sum) {
	if (x == n && y == m) {            // ç»ˆç‚¹ 
		ans = max(ans, sum+a[x][y]);
		return ;
	}
	for (int i = 0; i < 3; i++) {
		int nx = x + dx[i], ny = y + dy[i];
		if (nx < 1 || ny < 1 || nx > n || ny > m) continue;
		if (vis[nx][ny]) continue ;
		vis[nx][ny] = 1;
		dfs(nx, ny, sum + a[x][y]);
		vis[nx][ny] = 0;
	}
}
int main()
{
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= m; j++) {
			scanf("%d", &a[i][j]);
			vis[i][j] = false;
		}
	}
	
	dfs(1, 1, 0);
	
	printf("%d\n", ans);
	return 0;
}
